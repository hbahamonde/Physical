  \hbox to 0pt{%
    \beamer@tempdim=-\beamer@vboxoffset%

    \advance\beamer@tempdim by -.05cm%

      \global\beamer@section@min@dim\beamer@tempdim
      \hbox{\beamer@link(#4){%
          \usebeamerfont{mini frame}%
          \ifnum\c@section>#1%
            %\usebeamercolor[fg]{mini frame}%
            %\usebeamertemplate{mini frame}%
            \usebeamercolor{mini frame}%

          \else%
            \ifnum\c@section=#1%
              \ifnum\c@subsection>#2%
                \usebeamercolor[fg]{mini frame}%
                \usebeamertemplate{mini frame}%
              \else%
                \ifnum\c@subsection=#2%
                  \usebeamercolor[fg]{mini frame}%

                    \usebeamertemplate{mini frame}%

                \fi%

            \else%

          \fi%

  }\fi\fi%

\begin{frame}[c]{}
  \begin{itemize}
    \item {\bf Motivate the problem}: It's clear that {\bf the better the candidate's looks, the higher the turnout}.

    \item {\bf Problem}: The literature \emph{only} looks at candidate attractiveness, which is just one dimension of physical appearance.

    \item {\bf Gaps}: we explore the degree in which a candidate's {\bf occupation is congruent with his/her physical appearance}.

    \item {\bf Empirics}: we exploit a novel data set of candidate's physical appearance in the context of the 2017 Finnish Municipal Elections.

    \item {\bf Results}: we find that there exists a systematic electoral penalty for female candidates that look-like and also hold working-class occupations.
  \end{itemize}
\end{frame}

\begin{frame}[c]{Milestones}
    \begin{itemize}
      \item {\bf Motivate the problem}: vote buying literature is purely based on the Expected Utility Theory (EUT) {\tiny (von Neumann and Morgenstern)}.
      \item {\bf Problem}: as a consequence, there are too many important loose ends.
      \item {\bf Propose a possible solution}: re-think about how parties make decisions under risk (Prospect Theory).
      \item {\bf Empirics}: we formalized a vote buying game, and then test it in an economic lab experiment.
      \item {\bf Results}: we find strong support in favor of Prospect Theory.
      \item {\bf Feedback wanted!}
    \end{itemize}
\end{frame}
